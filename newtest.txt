---
- name: Disable HTTP/2 on Windows
  hosts: windows
  gather_facts: no
  tasks:
    - name: Disable HTTP/2 for HTTP (port 80)
      win_regedit:
        path: HKLM:\SOFTWARE\Microsoft\IIS\Parameters
        name: EnableHttp2
        value: 0
        type: dword
        state: present
    
    - name: Disable HTTP/2 for HTTPS (port 443)
      win_regedit:
        path: HKLM:\SOFTWARE\Microsoft\IIS\Parameters
        name: EnableHttp2
        value: 0
        type: dword
        state: present
    
    - name: Reboot the server to apply changes
      win_reboot:
        reboot_timeout: 600
